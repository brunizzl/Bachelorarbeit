\section{Definitionen}
\begin{frame}[fragile]{Term}
	\begin{block}{Definition}
		Terme sind
		\begin{itemize}
			\item{Konstantensymbole, etwa \Mono{1}, \Mono{-3i}, \Mono{a}, \Mono{sin}}
			\item{oder Funktionsanwendungen eines Terms $f$ auf die Terme $t_0, \dots, t_n$, geschrieben
			$f(t_1, \dots t_n)$, etwa \Mono{sin(3)}}
		\end{itemize}
	\end{block}		
	
	\begin{columns}[t] %t für align nach
